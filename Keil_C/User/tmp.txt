void TIMER0_IRQHandler(void)
{
if(TIM_GetFlagStatus(TIM0, TIM_Flag_TI) == SET)
  {

    TIM_ClearFlag(TIM0, TIM_Flag_TI);//清除TIM0中断
    TIM0_count++;//自定义TIM0计数变量加1（TIM0做定时器，定时器频率为1000Hz,每过1ms进入一次定时器中断，每进入一次定时器中断，该计数变量加1）
  }

}

void TIMER4_5_IRQHandler(void)
{

if(TIM_GetFlagStatus(TIM5, TIM_Flag_TI|TIM_Flag_EXR|TIM_Flag_EXF)==SET)
{
	if (GPIO_ReadDataBit(GPIOA, GPIO_Pin_14) == 1)
		{
			if (GPIO_ReadDataBit(GPIOA, GPIO_Pin_13) == 1)
			{
				Encoder_Count --;
			}
		}

}
if(TIM_GetFlagStatus(TIM5, TIM_Flag_TI|TIM_Flag_EXR|TIM_Flag_EXF)==SET)
{
	if (GPIO_ReadDataBit(GPIOA, GPIO_Pin_14) == 0)
		{
			if (GPIO_ReadDataBit(GPIOA, GPIO_Pin_13) == 0)
			{
				Encoder_Count--;
			}
		}

}
	TIM_ClearFlag(TIM5, TIM_Flag_TI|TIM_Flag_EXR|TIM_Flag_EXF);//Generated by EasyCodeCube, forbid editing!!!
}

void TIMER6_7_IRQHandler(void)
{

if(TIM_GetFlagStatus(TIM6, TIM_Flag_TI|TIM_Flag_EXR|TIM_Flag_EXF)==SET)
{
	if (GPIO_ReadDataBit(GPIOA, GPIO_Pin_13) == 1)
		{
			if (GPIO_ReadDataBit(GPIOA, GPIO_Pin_14) == 1)
			{
				Encoder_Count++;
			}
		}

}
if(TIM_GetFlagStatus(TIM6, TIM_Flag_TI|TIM_Flag_EXR|TIM_Flag_EXF)==SET)
{
	if (GPIO_ReadDataBit(GPIOA, GPIO_Pin_13) == 0)
		{
			if (GPIO_ReadDataBit(GPIOA, GPIO_Pin_14) == 0)
			{
				Encoder_Count++;
			}
		}

}

	TIM_ClearFlag(TIM6, TIM_Flag_TI|TIM_Flag_EXR|TIM_Flag_EXF);//Generated by EasyCodeCube, forbid editing!!!
}
